obj.create()

    miao: 'miao'
};

this: { a: 'a1', b: 'b2', c: 'c3' }
obj.forEach(fn)

    a: 'a1',
    b: 'b2',
    c: 'c3'
}
var out = [];
var indexs = [];
o.forEach(function(item, index) {
    out.push(item);
    indexs.push(index);

Report: function () { [native code] } this: { a: 'a1', b: 'b2', c: 'c3' }
Object.forEach(fn)
var o = {
    a: 'a1',
    b: 'b2',
    c: 'c3'
}
var out = [];
var indexs = [];
report(Object.forEach);

    out.push(item);
    indexs.push(index);

assert.deepEqual(['a', 'b', 'c'], indexs);
